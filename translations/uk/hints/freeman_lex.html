from re import sub, fullmatch


def long_pressed(text: str, typed: str) -> bool:

    pattern = sub(r'(\S)', r'\1+', text)
    pattern = sub(r'([\'"/?!.,;])', r'\\\1', pattern)

    return bool(fullmatch(pattern, typed)) and text!= typed
